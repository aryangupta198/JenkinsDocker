def builtDockerImage



pipeline{
agent any



stages{



stage('Clone repo') {



           steps {



               git url: 'https://github.com/aryangupta198/JenkinsDocker.git', branch: 'main'



           }



       }





      stage('Build image') {



           steps {



               script {



                   builtDockerImage = docker.build "aryan198g/jenkins-docker:v1.0"



               }



           }



       }



      



       stage('Push image') {



           steps {



               script {



                       docker.withRegistry('', 'docker-cred-aryan') {



                           builtDockerImage.push()



                       }



               }



           }



       }



   }



}
